\begin{equation*}
\begin{split}
\nonumber 
      && f(x) = \frac{2x}{4x - 7},
          x \ne \frac{7}{4}\\
\Rightarrow &&         y &= \frac{2x}{4x - 7}\\
\Rightarrow &&  4xy - 7y &= 2x\\
\Rightarrow &&  4xy - 2x &= 7y\\
\Rightarrow && x(4y - 2) &= 7y\\
\Rightarrow &&         x &= \frac{7y}{4y - 2}\\
\therefore
 && f(x)^{-1} = \frac{7x}{4x - 2},
          x \ne \frac{1}{2}\\
\end{split}
\end{equation*}
